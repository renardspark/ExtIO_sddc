80/*
81 * This function uploads the firmware from the given file into RAM.
82 * Stage == 0 means this is a single stage load (or the first of
83 * two stages). Otherwise it's the second of two stages; the
84 * caller having preloaded the second stage loader.
85 *
86 * The target processor is reset at the end of this upload.
87 */
88extern int ezusb_load_ram(libusb_device_handle *device,
89 const char *path, int fx_type, int img_type, int stage);

93/*
94 * This function uploads the firmware from the given file into EEPROM.
95 * This uses the right CPUCS address to terminate the EEPROM load with
96 * a reset command where FX parts behave differently than FX2 ones.
97 * The configuration byte is as provided here (zero for an21xx parts)
98 * and the EEPROM type is set so that the microcontroller will boot
99 * from it.
100 *
101 * The caller must have preloaded a second stage loader that knows
102 * how to respond to the EEPROM write request.
103 */
104extern int ezusb_load_eeprom(libusb_device_handle *device,
105 const char *path, int fx_type, int img_type, int config);
